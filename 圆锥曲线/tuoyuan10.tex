\documentclass{standalone}
\usepackage{tkz-base}
\usepackage{tkz-fct}
\usepackage{tkz-euclide}
\usepackage{tikz}
\begin{document}
    \begin{tikzpicture}
        \pgfmathsetmacro\x{3}
        \pgfmathsetmacro\y{2.7}
        \pgfmathsetmacro\a{3}
        \pgfmathsetmacro\b{sqrt(6)}
        \pgfmathsetmacro\c{sqrt(abs(\a^2-\b^2))}
        \pgfmathsetmacro\px{1}
        \pgfmathsetmacro\py{sqrt((\b^2)*(1-((\px)^2)/(\a^2)))}
        \pgfmathsetmacro\rPF{sqrt((\px+\c)^2+\py^2)/2}
        \pgfmathsetmacro\RPF{sqrt((\px-\c)^2+\py^2)/2}
        \pgfmathsetmacro\e{\a^2/\c}
        \coordinate (O) at (0,0);
        \coordinate (F1) at (-\c,0);
        \coordinate (F2) at (\c,0);
        \coordinate (E) at (\e,0);
        \draw[-stealth] (-\x,0) -- (\e+0.3,0) node [below] {$x$};
        \draw[-stealth] (0,-\y) -- (0,\y) node [right] {$y$};
        \draw[domain=-2:\e+0.1,name path =a ] plot (\x,{-0.47*(\x-\e)});
        \draw[name path = b ] (O) circle [x radius=\a,y radius=\b];
        \fill [name intersections={of=a and b,by={A,B}}]
        (A) circle (1pt) node [below left] {$A$}
        (B) circle (1pt) node [below] {$B$};
        \fill (O) node [below left] {$O$} circle (1pt);
        \fill (F1) node [below] {$F_1$} circle (1pt);
        \fill (F2) node [below] {$F_2$} circle (1pt);
        \fill (E) node [above] {$E$} circle (1pt);
        \draw[cyan,dashed] (F1) -- (A) (F2)--(B);
    \end{tikzpicture}
\end{document}
