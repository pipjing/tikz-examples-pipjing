\begin{tikzpicture}
\pgfmathsetmacro\x{3}
\pgfmathsetmacro\y{2.7}
\pgfmathsetmacro\a{3}
\pgfmathsetmacro\b{2}
\pgfmathsetmacro\c{sqrt(abs(\a^2-\b^2))}
\coordinate (O) at (0,0);
\coordinate (F1) at (-\c,0);
\coordinate (F2) at (\c,0);
\draw[-stealth] (-\x,0) -- (\x+0.3,0) node [below] {$x$};
\draw[-stealth] (0,-\y) -- (0,\y) node [right] {$y$};
\draw[name path = a ,domain = 1.5:2.7] plot (\x, {3.4*(\x-\c)});
\draw[name path = b ] (O) circle [x radius=\a,y radius=\b];
\fill [name intersections={of=a and b,by={A,B}}]
(A) circle (1pt) node [below left] {$A$}
(B) circle (1pt) node [below left] {$B$};
\fill (O) node [below left] {$O$} circle (1pt);
\fill (F1) node [below] {$F_1$} circle (1pt);
\fill (F2) node [below] {$F_2$} circle (1pt);
\draw[cyan,dashed,thick] (B)--(F1)node[above,pos=0.6]{$2a-x$} -- (A)node[left,pos=0.6]{$2a-2x$};
\path (B)--(F2)node[left,pos=0.6]{$x$} --(A)node[left,pos=0.6]{$2x$};
 \end{tikzpicture}
